CREATE TABLE IF NOT EXISTS public.ai_tasks (

  created_at      timestamptz NOT NULL DEFAULT now(),
  updated_at      timestamptz NOT NULL DEFAULT now(),
  task_type       text        NOT NULL,
  title           text        NOT NULL,
  description     text,
  priority        public.ai_task_priority NOT NULL DEFAULT 'media',
  status          public.ai_task_status   NOT NULL DEFAULT 'pendente',
  entity_type     text,
  entity_id       uuid,

  sale_id         uuid REFERENCES public.sales(id)        ON DELETE SET NULL,
  source          text        NOT NULL DEFAULT 'agente_ia',
  auto_executable boolean     NOT NULL DEFAULT false,
  payload         jsonb,
  result          jsonb,
  assigned_to     uuid REFERENCES public.hub_users(id)    ON DELETE SET NULL,
  reviewed_by     uuid REFERENCES public.hub_users(id)    ON DELETE SET NULL,

);
