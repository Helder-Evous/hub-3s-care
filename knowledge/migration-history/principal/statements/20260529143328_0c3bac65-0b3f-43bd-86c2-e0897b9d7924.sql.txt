-- Enums
CREATE TYPE public.clinic_status AS ENUM ('ativa','pausada','cancelada')
CREATE TYPE public.channel_type AS ENUM ('api_oficial','whatsapp_comum','numero_reserva','api_reserva')
CREATE TYPE public.channel_status AS ENUM ('ativo','em_atencao','critico','livre','bloqueado','desconectado')
CREATE TYPE public.alert_severity AS ENUM ('baixa','media','alta','critica')
CREATE TYPE public.alert_status AS ENUM ('aberto','em_tratamento','resolvido')
CREATE TYPE public.data_origin AS ENUM ('meta_api','plataforma','manual','robo_tela')
CREATE TYPE public.quality_rating AS ENUM ('alto','medio','baixo','desconhecido')
CREATE TYPE public.number_status AS ENUM ('conectado','restrito','banido','pendente','desconhecido')
-- Clinics
CREATE TABLE public.clinics (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT NOT NULL,
  city TEXT,
  state TEXT,
  responsible TEXT,
  status public.clinic_status NOT NULL DEFAULT 'ativa',
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
)
-- Channels
CREATE TABLE public.channels (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  clinic_id UUID NOT NULL REFERENCES public.clinics(id) ON DELETE CASCADE,
  phone_number TEXT NOT NULL,
  channel_type public.channel_type NOT NULL,
  status public.channel_status NOT NULL DEFAULT 'ativo',
  activated_at DATE,
  carrier TEXT,
  device TEXT,
  responsible TEXT,
  notes TEXT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
)
-- Channel monitoring
CREATE TABLE public.channel_monitoring (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  channel_id UUID NOT NULL REFERENCES public.channels(id) ON DELETE CASCADE,
  connected BOOLEAN,
  quality_rating public.quality_rating,
  number_status public.number_status,
  message_limit INT,
  has_restriction BOOLEAN,
  qr_disconnected BOOLEAN,
  session_expired BOOLEAN,
  receiving_messages BOOLEAN,
  sending_messages BOOLEAN,
  last_message_received TIMESTAMPTZ,
  last_message_sent TIMESTAMPTZ,
  last_check TIMESTAMPTZ NOT NULL DEFAULT now(),
  data_origin public.data_origin NOT NULL DEFAULT 'manual',
  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
)
-- Contingency items
CREATE TABLE public.contingency_items (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  clinic_id UUID NOT NULL UNIQUE REFERENCES public.clinics(id) ON DELETE CASCADE,
  has_backup_number BOOLEAN NOT NULL DEFAULT false,
  has_backup_api BOOLEAN NOT NULL DEFAULT false,
  has_approved_template BOOLEAN NOT NULL DEFAULT false,
  has_backup_chip BOOLEAN NOT NULL DEFAULT false,
  has_backup_device BOOLEAN NOT NULL DEFAULT false,
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
)
-- Alerts
CREATE TABLE public.alerts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  clinic_id UUID REFERENCES public.clinics(id) ON DELETE CASCADE,
  channel_id UUID REFERENCES public.channels(id) ON DELETE SET NULL,
  alert_type TEXT NOT NULL,
  severity public.alert_severity NOT NULL,
  message TEXT NOT NULL,
  status public.alert_status NOT NULL DEFAULT 'aberto',
  responsible TEXT,
  notes TEXT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  resolved_at TIMESTAMPTZ
)
-- Audit logs
CREATE TABLE public.audit_logs (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  entity TEXT NOT NULL,
  entity_id UUID,
  action TEXT NOT NULL,
  actor TEXT,
  payload JSONB,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
)
